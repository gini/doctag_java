18  /**
19   * A docTag connector which handles reading and writing docTag meta data in PDF documents using the
20   * <a href="http://pdfbox.apache.org/">Apache PDFBox</a> library.
21   *
22   * @author Jochen Schalanda

24  public class PdfConnector extends Connector {

26      /**
27       * The name of the property field inside the PDF document which contains the docTag meta data.
28       */
29      private static final String DOCTAG_FIELD_NAME = "doctag";
30  
31      private PDDocument pdfDocument;
32      private final Parser parser;
33      private final Generator generator;
34  
35      /**
36       * Create a new instance of PdfConnector.
37       *
38       * @param inputStream The input stream containing the PDF data which will be processed
39       * @throws ConnectorException If the input stream couldn't be read or doesn't contain a PDF document
40       */
41      public PdfConnector(final InputStream inputStream) throws ConnectorException {
42          this(inputStream, new Parser(), new Generator());
43      }

45      /**
46       * Create a new instance of PdfConnector.
47       *
48       * @param inputStream The input stream containing the PDF data which will be processed
49       * @param parser      The parser which is being used to parse the docTag meta data from the JSON format
50       * @param generator   The generator which is being used to generate the docTag meta data JSON format
51       * @throws ConnectorException If the input stream couldn't be read or doesn't contain a PDF document
52       */
53      public PdfConnector(final InputStream inputStream, final Parser parser, final Generator generator)
54              throws ConnectorException {
55  
56          super(inputStream);
57  
58          assert parser != null;
59          assert generator != null;
60  
61          this.parser = parser;
62          this.generator = generator;
63  
64          try {
65              this.pdfDocument = PDDocument.load(this.inputStream);
66          } catch (IOException e) {
67              throw new ConnectorException("Couldn't load PDF document from input stream", e);
68          }
69      }
70  
71      @Override
72      /**
73       * Read data from the input stream, parse it, and return a new docTag meta data document.
74       *
75       * @return The parsed docTag meta data
76       * @throws DataNotFoundException If the input stream doesn't contain any docTag meta data
77       * @throws InvalidDataException  If the input stream doesn't contain any valid docTag meta data
78       */
79      public Document read() throws DataNotFoundException, InvalidDataException {
80  
81          Document result;
82  
83          PDDocumentInformation documentInformation = pdfDocument.getDocumentInformation();
84          String embeddedDocTag = documentInformation.getCustomMetadataValue(DOCTAG_FIELD_NAME);
85  
86          if (null == embeddedDocTag) {
87  
88              throw new DataNotFoundException("Couldn't find meta data field " + DOCTAG_FIELD_NAME);
89          }
90  
91          InputStream embeddedDocTagStream = new ByteArrayInputStream(embeddedDocTag.getBytes());
92  
93          try {
94              result = parser.parse(embeddedDocTagStream);
95          } catch (IOException e) {
96              throw new InvalidDataException("Couldn't parse embedded docTag data", e);
97          }
98  
99          return result;
100     }
101 
102     @Override
103     /**
104      * Write the data read from the input stream back into an output stream while updating the docTag meta data inside.
105      *
106      * @param outputStream The {@link OutputStream} in which the modified original data is being written
107      * @param document     The docTag meta data document which should be written into the originally read data
108      * @throws ConnectorException   If an error occurred while writing to the output stream
109      * @throws InvalidDataException if the docTag meta data document contains invalid data
110      */
111     public void write(final OutputStream outputStream, final Document document)
112             throws ConnectorException, InvalidDataException {
113 
114         String docTagJson;
115 
116         try {
117             docTagJson = generator.generate(document);
118 
119         } catch (IOException e) {
120             throw new InvalidDataException("Couldn't generate docTag data", e);
121         }
122 
123         PDDocumentInformation documentInformation = pdfDocument.getDocumentInformation();
124         documentInformation.setCustomMetadataValue(DOCTAG_FIELD_NAME, docTagJson);
125 
126         try {
127             pdfDocument.save(outputStream);
128         } catch (IOException e) {
129             throw new ConnectorException("Couldn't write PDF document to output stream", e);
130         } catch (COSVisitorException e) {
131             throw new ConnectorException("Corrupt PDF data", e);
132         }
133     }
134 
135     @Override
136     protected void finalize() throws Throwable {
137

139 
140         super.finalize();
141     }
